## рдкреВрд░реНрд╡-рдкрд░рд┐рдирд┐рдпреЛрдЬрди рдХреНрд╖рдорддрд╛ рдЬрд╛рдВрдЪ

### рд╕реНрд╡рдЪрд╛рд▓рд┐рдд рдХреНрд╖рдорддрд╛ рд╕рддреНрдпрд╛рдкрди рд╕реНрдХреНрд░рд┐рдкреНрдЯ
```bash
#!/bin/bash
# capacity-check.sh - рдкрд░рд┐рдирд┐рдпреЛрдЬрди рд╕реЗ рдкрд╣рд▓реЗ Azure рдХреНрд╖рдорддрд╛ рдХреЛ рдорд╛рдиреНрдп рдХрд░реЗрдВ

set -e

LOCATION=${1:-eastus2}
SUBSCRIPTION_ID=$(az account show --query id -o tsv)

echo "Checking Azure capacity for location: $LOCATION"
echo "Subscription: $SUBSCRIPTION_ID"
echo "======================================================"

# рдХреЛрдЯрд╛ рдЙрдкрдпреЛрдЧ рдХреА рдЬрд╛рдВрдЪ рдХрд░рдиреЗ рдХреЗ рд▓рд┐рдП рдлрд╝рдВрдХреНрд╢рди
check_quota() {
    local resource_type=$1
    local required=$2
    
    echo "Checking $resource_type quota..."
    
    case $resource_type in
        "compute")
            usage=$(az vm list-usage --location "$LOCATION" --query "[?localName=='Total Regional vCPUs'].{current:currentValue,limit:limit}" -o json)
            current=$(echo "$usage" | jq -r '.[0].current')
            limit=$(echo "$usage" | jq -r '.[0].limit')
            ;;
        "storage")
            usage=$(az storage account show-usage --query "{current:value,limit:limit}" -o json)
            current=$(echo "$usage" | jq -r '.current')
            limit=$(echo "$usage" | jq -r '.limit')
            ;;
        "network")
            usage=$(az network list-usages --location "$LOCATION" --query "[?localName=='Virtual Networks'].{current:currentValue,limit:limit}" -o json)
            current=$(echo "$usage" | jq -r '.[0].current')
            limit=$(echo "$usage" | jq -r '.[0].limit')
            ;;
    esac
    
    available=$((limit - current))
    
    if [ "$available" -ge "$required" ]; then
        echo "тЬЕ $resource_type: $available/$limit available (need $required)"
    else
        echo "тЭМ $resource_type: Only $available/$limit available (need $required)"
        return 1
    fi
}

# рд╡рд┐рднрд┐рдиреНрди рд╕рдВрд╕рд╛рдзрди рдХреЛрдЯрд╛ рдХреА рдЬрд╛рдВрдЪ рдХрд░реЗрдВ
check_quota "compute" 4      # 4 vCPUs рдХреА рдЖрд╡рд╢реНрдпрдХрддрд╛ рд╣реИ
check_quota "storage" 2      # 2 рд╕реНрдЯреЛрд░реЗрдЬ рдЦрд╛рддреЛрдВ рдХреА рдЖрд╡рд╢реНрдпрдХрддрд╛ рд╣реИ
check_quota "network" 1      # 1 рд╡рд░реНрдЪреБрдЕрд▓ рдиреЗрдЯрд╡рд░реНрдХ рдХреА рдЖрд╡рд╢реНрдпрдХрддрд╛ рд╣реИ

echo "======================================================"
echo "тЬЕ Capacity check completed successfully!"
```

check_app_service_capacity() {

    local sku=$2
    
    echo "Checking App Service Plan capacity for $sku in $location"
    
    # рдХреНрд╖реЗрддреНрд░ рдореЗрдВ рдЙрдкрд▓рдмреНрдз SKU рдХреА рдЬрд╛рдВрдЪ рдХрд░реЗрдВ

    if [ -n "$available_skus" ]; then
