resource webApp 'Microsoft.Web/sites@2022-03-01' = {
  name: '${applicationName}-web-${resourceToken}'
  location: location
  tags: tags
  properties: {

          value: 'false'

          name: 'NODE_ENV'
          value: 'production'
        }
      ]

  }
